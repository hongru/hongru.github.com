 27 	Class.extend = function (prop) {
 28 		var _super = this.prototype;
 29 		//创建一个实例，但不执行init
 30 		initializing = true;
 31 		var prototype = new this();
 32 		initializing = false;
 33 
 34 		for (var name in prop) {
 35 			// 用闭包保证多级继承不会污染
 36 			prototype[name] = (typeof prop[name] === 'function' && typeof _super[name] === 'function' && superTest.test(prop[name])) ? (function (name, fn) {
 37 					return function () {
 38 						var temp = this._super;	
 39 						// 当前子类通过_super继承父类
 40 						this._super = _super[name];
 41 						//继承方法执行完毕后还原
 42 						var ret = fn.apply(this, arguments);
 43 						this._super = temp;
 44 
 45 						return ret;
 46 					}
 47 				})(name, prop[name]) : prop[name];
 48 		}
 49 		
 50 		//真实的constructor
 51 		function Class () {
 52 			if (!initializing && this.init) {
 53 				this.init.apply(this, arguments);
 54 			}
 55 		}
 56 		Class.prototype = prototype;
 57 		Class.constructor = Class;
 58 		Class.extend = arguments.callee;
 59 
 60 		return Class;
 61 	}

106 						return;
107 					}

109 						// 删除指定的监听器
110 						if (this.eventListener[type][i] == func) {
111 							delete this.eventListener[type][i];
112 							this.eventListener[type].splice(i, 1);
113 						}
114 					}
115 					// 如果这种类型没有监听器，删除它
116 					if (this.eventListener[type].length === 0) {
117 						delete this.eventListener[type];
118 					}
119 				},
120 				removeAllEventListener: function (type) {
121 					if (this.eventListener[type] === null || this.eventListener[type] === undefined) {
122 						return;
123 					}	
124 					this.eventListener[type].splice();
125 					delete this.eventListener[type];
126 				},
127 				hasEventListener: function (type) {
128 					return (!!this.eventListener[type] && this.eventListener[type].length > 0);

232 					var mouseX = x, mouseY = y;
233 					var _hoverChildren = [];
234                     // check isMouseover
235                     function isMouseover (child) {
236                         var ret = false;
237                         // checkType  rect|circle|poly
238                         if (!child.checkType) {
239                             child.checkType = 'rect';
240                         }
241                         switch (child.checkType) {
242                             case 'rect':
243                                 ret = (mouseX > child.x 
244                                         && mouseX < child.x + child.width
245                                         && mouseY > child.y
246                                         && mouseY < child.y + child.height);
247                                 break;
248                             case 'circle':
249                                 if (typeof child.radius != 'number') { throw 'no radius or radius is not a number' }
250                                 ret = (Math.sqrt(Math.pow((mouseX-child.x) ,2) + Math.pow((mouseY-child.y) ,2)) < child.radius);
251                                 break;
252                             case 'poly':
253                                 // to be continue...
254                                 break;
255                         }
256                         return ret;
257                     }
258                     
259 					for (var i=0; i<this.children.length; i++) { 
260                         var child = this.children[i];
261 						if (!!child.dispatchMouseEvent) {
262 							child.dispatchMouseEvent(type, mouseX-child.x, mouseY-child.y);
263 						}
264 						//鼠标悬浮于子对象上面
265 						if (isMouseover(child)) {
266 							type == 'mousemove' && _hoverChildren.push(child);
267 						
268                             if (child.eventListener[type] == null
269                                 || child.eventListener[type] == undefined) {
270                                 continue; // 没有事件监听器
271                             }
272                             // 有事件监听则遍历执行
273                             for (var j=0, arr=child.eventListener[type]; j < arr.length; j++) {
274                                 arr[j](mouseX-child.x, mouseY-child.y);
275                             }
276 						}
277 					};
278 					if (type != 'mousemove') {
279 						return; // 不是 mousemove事件则到此结束
280 					}
281 					// 以下是处理mousemove事件
282 					for (var k=0; k<this.hoverChildren.length; k++) {
283 						// 原来hoverChildren中有的，现在没有的，转而执行 mouseout
284 						var has = false;
285 						for (var m=0; m<_hoverChildren.length; m++) {
286 							if (this.hoverChildren[k] == _hoverChildren[m]) {
287 								has = true;
288 							}
289 						}
290 						if (!has) {
291 							//不存在了，处理 this.hoverChildren[k] 的mouseout
292 							// 刚好又有事件在监听mouseout，则执行
293 							if (!!this.hoverChildren[k].eventListener['mouseout']) {
294 								for (var i=0, outObj = this.hoverChildren[k]; i<outObj.eventListener['mouseout'].length; i++) {
295 									outObj.eventListener['mouseout'][i](mouseX-outObj.x, mouseY-outObj.y);
296 								}
297 							}
298 							// 处理完后就销毁
299 							delete this.hoverChildren[k];
300 						}
301 					};
302 					// 原来hoverChildren中没有的，现在有了，证明mouseover
303 					for (var k=0; k<_hoverChildren.length; k++) {
304 						var has = false;
305 						for (var m=0; m<this.hoverChildren.length; m++) {
306 							if (_hoverChildren[k] == this.hoverChildren[m]) {
307 								has = true;
308 							}
309 						};
310 						if (!has) {
311 							//证明鼠标刚进入，处理mouseenter或mouseover
312 							this.hoverChildren.push(_hoverChildren[k]);
313 							if (_hoverChildren[k].eventListener['mouseover']) {
314 								for (var i=0, enterObj = _hoverChildren[k]; i<enterObj.eventListener['mouseover'].length; i++) {

857 	 * 粒子系统，相当于粒子的一个collection

859      * @class 
860      * @name ParticleSystem
861      * @memberOf Laro
862      * 
863      * @return ParticleSystem 实例
864 
865 	 */
866 	var ParticleSystem = Class.extend({
867 		init: function () {
868 			this.$private = {
869 				particles : []
870 			}
871 			
872 			this.gravity = new Vector2(0, 100);
873 			this.effectors = [];
874 
875 		},
876 		// push 粒子到发射备用区
877 		emit: function (particle) {
878 			this.$private.particles.push(particle);
879 		},
880 		// 模拟运动(在当前时间微分下)
881 		simulate: function (dt) {
882 			this.aging(dt);
883 			this.applyGravity();
884 			this.applyEffectors();
885 			this.kinematics(dt);
886 		},
887 		// 判断粒子的生存时间
888 		aging: function (dt) {
889 			for (var i=0; i < this.$private.particles.length; ) {
890 				var p = this.$private.particles[i];
891 				p.age += dt;
892 				if (p.age > p.life) {
893 					this.kill(i);
894 				} else  {
895 					i ++;
896 				}
897 			}
898 		},
899 		kill: function (index) {
900 			if (index < this.$private.particles.length) {
901 				this.$private.particles.splice(index, 1);
902 			}	  
903 		},
904 		applyGravity: function () {
905 			for (var i in this.$private.particles) {
906 				this.$private.particles[i].acceleration = this.gravity;
907 			}			  
908 		},
909 		applyEffectors: function () {
910 			for (var j in this.effectors) {
911 				var apply = this.effectors[j].apply;
912 				for (var i in this.$private.particles) {
913 					apply(this.$private.particles[i]);
914 				}
915 			}				
916 		},
917 		// 运动学变换，矢量叠加
918 		kinematics: function (dt) {
919 			for (var i in this.$private.particles) {
920 				var p  = this.$private.particles[i];
921 				p.position = p.position.add(p.velocity.multiply(dt));
922 				p.velocity = p.velocity.add(p.acceleration.multiply(dt));
923 			}
924 		},
925 		/**
926 		 * 默认粒子的寿命由透明度表示
927 		 */
928 		render: function (ctx) {
929 			for (var i in this.$private.particles) {
930 				var p  = this.$private.particles[i],
931 					alpha = 1- (p.age/p.life);
932 				ctx.fillStyle = 'rgba('+Math.floor(p.color.r*255)+', '+Math.floor(p.color.g*255)+', '+Math.floor(p.color.b)+', '+alpha.toFixed(2)+')';
933 				ctx.beginPath();
934 				ctx.arc(p.position.x, p.position.y, p.size, 0, Math.PI*2, true);
935 				ctx.closePath();
936 				ctx.fill();
937 			}
